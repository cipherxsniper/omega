// 🌱 ECOSYSTEM FORMATION RULE (frontend concept)

function assignEcosystem(p, ecosystems) {

  let target = null;

  for (let e of ecosystems) {
    const dx = e.x - p.x;
    const dy = e.y - p.y;

    if (Math.sqrt(dx*dx + dy*dy) < 0.15) {
      target = e;
      break;
    }
  }

  if (!target) {
    target = {
      id: Math.random(),
      x: p.x,
      y: p.y,
      particles: 0,
      energy: 1
    };
    ecosystems.push(target);
  }

  target.particles++;
  return target.id;
}
